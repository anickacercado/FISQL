CREAR FUNCION morir () text {
     RETORNO "Algo sali mal, no te puedo seguir calificando... :-( ";

CREAR FUNCION sumar (INTEGER @a, INTEGER @b) INTEGER {
    RETORNO @a + @b;
}

CREAR FUNCION restar (INTEGER @a, INTEGER @b) INTEGER {
    DECLARAR @resultado INTEGER=0;
    @resultado = @a - @b;
    RETORNO @resultado;
}

CREAR FUNCION multiplicar (INTEGER @a, INTEGER @b) INTEGER {
    RETORNO @a * @b;
}

CREAR FUNCION dividir (INTEGER @a, INTEGER @b) INTEGER {
    DECLARAR @resultado INTEGER=0;
    @resultado = @a / @b;
    RETORNO @resultado;
}

CREAR FUNCION potencia (INTEGER @a, INTEGER @b) INTEGER {
    RETORNO @a ^ @b;
}
